<script>
// ========== 已帮你填好关键配置 ==========
const CORP_ID = "ding1c9e53a8cf90f15ff2c783f7214b6d69"; // 你的钉钉企业ID
const TARGET_URL = "https://arena.ai/"; // 目标网站地址

// 禁用右键、禁用复制，进一步防止地址泄露
document.addEventListener('contextmenu', e => e.preventDefault()); // 禁用右键菜单
document.addEventListener('copy', e => e.preventDefault()); // 禁用复制功能

// 1. 首先判断是否在钉钉内打开
if (!dd.env || !dd.env.platform.isDingTalk) {
  document.getElementById("tip").innerText = "禁止外部访问，请在钉钉内打开";
} else {
  // 2. 钉钉环境初始化完成后，获取免登码校验身份
  dd.ready(function () {
    dd.runtime.permission.requestAuthCode({
      corpId: CORP_ID,
      // 身份校验成功：显示目标网站
      onSuccess: function (res) {
        document.getElementById("tip").style.display = "none";
        const iframe = document.getElementById("iframe");
        iframe.src = TARGET_URL;
        iframe.style.display = "block";
      },
      // 身份校验失败：提示无权限
      onFail: function (err) {
        document.getElementById("tip").innerText = "无访问权限，请使用企业钉钉账号打开";
      }
    });
  });
}
</script>
